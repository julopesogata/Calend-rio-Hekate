INICIALIZAÇÃO DO CALENDÁRIO const calendarEl = document.getElementById('calendar'); const calendar = new FullCalendar.Calendar(calendarEl, { initialView: 'dayGridMonth', locale: 'pt-br', headerToolbar: { left: 'prev,next', center: 'title', right: 'today' }, events: [...eventosFixos, ...fasesLua], eventDisplay: 'block', eventDidMount: function(info) { // Estilização dos eventos const tipo = info.event.extendedProps.tipo; if (tipo === 'festival') { info.el.style.color = '#9b59b6'; info.el.style.fontWeight = 'bold'; } else if (tipo === 'sabbat') { info.el.style.color = '#e74c3c'; info.el.style.fontWeight = 'bold'; } else if (tipo === 'lunar') { info.el.style.backgroundColor = 'transparent'; info.el.style.boxShadow = 'none'; info.el.style.cursor = 'default'; } }, datesSet: function(info) { atualizarEventosMes(info.start, info.end); } }); calendar.render(); // 4. ATUALIZAR EVENTOS DO MÊS (COM HORÁRIOS) function atualizarEventosMes(start, end) { const todosEventos = [...eventosFixos, ...fasesLua]; const eventosDoMes = todosEventos.filter(evento => { const dataEvento = new Date(evento.start); return dataEvento >= start && dataEvento <= end; }); // Ordenar por data eventosDoMes.sort((a, b) => new Date(a.start) - new Date(b.start)); const listaEventos = document.getElementById('eventos-lista'); listaEventos.innerHTML = ''; if (eventosDoMes.length === 0) { listaEventos.innerHTML = '

Nenhum evento importante este mês.

'; return; } eventosDoMes.forEach(evento => { const dataEvento = new Date(evento.start); const dataFormatada = dataEvento.toLocaleDateString('pt-BR', { day: '2-digit', month: 'long', year: 'numeric' }); // Formatar hora quando existir let horaFormatada = ''; if (evento.start.includes('T')) { horaFormatada = ' - ' + dataEvento.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }); } // Capitalizar primeira letra do mês const dataFormatadaCapitalizada = dataFormatada.replace(/^\w/, c => c.toUpperCase()); const eventoDiv = document.createElement('div'); eventoDiv.className = 'evento-item'; let tipoClass = ''; let tipoTexto = ''; switch(evento.tipo) { case 'festival': tipoClass = 'tipo-festival'; tipoTexto = 'Festival'; break; case 'sabbat': tipoClass = 'tipo-sabbat'; tipoTexto = 'Sabbat'; break; case 'lunar': tipoClass = 'tipo-lunar'; tipoTexto = 'Fase Lunar'; break; case 'deipnon': tipoClass = 'tipo-deipnon'; tipoTexto = 'Deipnon'; break; case 'noumenia': tipoClass = 'tipo-noumenia'; tipoTexto = 'Noumenia'; break; } eventoDiv.innerHTML = `
${dataFormatadaCapitalizada}${horaFormatada}
${tipoTexto} ${evento.nome || evento.title}
${evento.descricao}
`; listaEventos.appendChild(eventoDiv); }); } // Atualizar eventos do mês atual inicialmente atualizarEventosMes( calendar.view.activeStart, calendar.view.activeEnd ); });
